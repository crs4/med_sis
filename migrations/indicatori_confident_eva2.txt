-- Nutrient imbalance SOC Decline

CREATE VIEW c_n_ratio AS
SELECT
    id,
    (org_car / n_tot) AS c_n
FROM public.lab_data
WHERE org_car IS NOT NULL AND n_tot IS NOT NULL AND n_tot != 0;


-- Sodium exchangeable percentage

CREATE VIEW sodium_exchangeable_percentage AS
SELECT
    id,
    ((na / cec)*100) AS esp
FROM public.lab_data
WHERE na IS NOT NULL AND cec IS NOT NULL AND cec != 0;
    

-- Sodium adsorption ratio

CREATE VIEW sodium_adsorption_ratio AS
SELECT
    id,
    (sol_na / SQRT(sol_ca + sol_mg)) AS sar
FROM public.lab_data
WHERE sol_na IS NOT NULL AND sol_ca IS NOT NULL AND sol_mg IS NOT NULL AND (sol_ca + sol_mg) != 0;


-- Sodicity/salinity ratio

CREATE VIEW sodicity_salinity_ratio AS
SELECT
    id,
    (sar / el_cond) AS sar_elcond
FROM public.lab_data
WHERE sar IS NOT NULL AND el_cond IS NOT NULL AND el_cond != 0;

-- Soil erodibility by water *** 
You have to use two other tables (I added tables number to easily find it in the document itselfs):

    s (table IV) based on the structural size (cell BL5 of 'Layer descriptions' in the excel file that you sent me).
    p (table III) based on hy_cond column of 'Lab data'.

CREATE VIEW soil_erodibility_by_water AS
SELECT
    id,
    (2.1 * POWER(10, -4) * (12 - (org_car * 1.724 / 10)) * POWER((silt * (silt + sand)), 1.14) + 3.25 * (s-2) + 2.5* (p-3)) / (100 * 7.59) AS k
FROM public.lab_data
WHERE org_car IS NOT NULL AND silt IS NOT NULL AND sand IS NOT NULL AND s IS NOT NULL AND p IS NOT NULL;

CREATE OR REPLACE VIEW soil_erodibility_by_water AS
SELECT
   id,    
   (
      2.1 * POWER(10, -4) * (12 - (ld.org_car * 1.724 / 10)) * POWER((ld.silt * (ld.silt + ld.sand)), 1.14)+ 3.25 * (
            CASE
                WHEN lst.size1 = 'VF' THEN 1
                WHEN lst.size1 = 'FI' THEN 2
                WHEN lst.size1 IN ('ME', 'CO') THEN 3
                WHEN lst.size1 IN ('CO', 'EC') THEN 4
                ELSE NULL
            END - 2
        ) + 2.5 * (ld.hy_cond - 3)) / (100 * 7.59) AS k
FROM public.lab_data ld
JOIN public.layer_structure_types lst
    ON lst.layer_id = ld.id
WHERE
    ld.upper = 0
    AND ld.org_car IS NOT NULL
    AND ld.silt IS NOT NULL
    AND ld.sand IS NOT NULL
    AND lst.size1 IS NOT NULL
    AND ld.hy_cond IS NOT NULL;

-- Air capacity

CREATE VIEW air_capacity AS
SELECT
    id,
    (satur - field_cap) AS ac
FROM public.lab_data
WHERE satur IS NOT NULL AND field_cap IS NOT NULL;


-- Plant available water capacity

CREATE VIEW plant_available_water_capacity AS
SELECT
    id,
    (field_cap - wilting_p) AS pawc
FROM public.lab_data
WHERE field_cap IS NOT NULL AND wilting_p IS NOT NULL;


-- Relative field capacity

CREATE VIEW relative_field_capacity AS
SELECT
    id,
   (field_cap / satur) AS rfc
FROM public.lab_data
WHERE field_cap IS NOT NULL AND satur IS NOT NULL AND satur != 0;
